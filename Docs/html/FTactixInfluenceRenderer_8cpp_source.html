17FColor FTactixInfluenceRenderer::HeatColor(float T)
18{
19 T = FMath::Clamp(T, 0.0f, 1.0f);
20 // Blue (cold) → Cyan → Green → Yellow → Red (hot)
21 if (T < 0.25f)
22 {
23 const float F = T / 0.25f;
24 return FColor(0, static_cast<uint8>(F * 255), 255); // Blue → Cyan
25 }
26 if (T < 0.5f)
27 {
28 const float F = (T - 0.25f) / 0.25f;
29 return FColor(0, 255, static_cast<uint8>((1.0f - F) * 255)); // Cyan → Green
30 }
31 if (T < 0.75f)
32 {
33 const float F = (T - 0.5f) / 0.25f;
34 return FColor(static_cast<uint8>(F * 255), 255, 0); // Green → Yellow
35 }
36 {
37 const float F = (T - 0.75f) / 0.25f;
38 return FColor(255, static_cast<uint8>((1.0f - F) * 255), 0); // Yellow → Red
39 }
40}

42void FTactixInfluenceRenderer::DrawChannel(const UWorld* World,
43 const Tactix::FTactixInfluenceMap<4>& Map,
44 std::size_t Channel,
45 float ZHeight,
46 float MinValue,
47 float QuadThickness)
48{
49 if (!World || Channel >= 4) return;
50
51 const int32 W = Map.GetWidth();
52 const int32 H = Map.GetHeight();
53 const float HalfCell = Map.GetCellSize() * 0.5f;
54
55 for (int32 y = 0; y < H; ++y)
56 for (int32 x = 0; x < W; ++x)
57 {
58 const Tactix::FTactixVec2 Center = Map.CellCenter(x, y);
59 const float Value = Map.SampleNearest(Channel, Center);
60 if (Value <= MinValue) continue;
61
62 const FColor Color = HeatColor(FMath::Clamp(Value, 0.0f, 1.0f));
63 const FVector BoxCenter(Center.X, Center.Y, ZHeight);
64 const FVector BoxExtent(HalfCell * 0.9f, HalfCell * 0.9f, QuadThickness);
65
66 DrawDebugBox(World, BoxCenter, BoxExtent, FQuat::Identity, Color,
67 false, 0.0f, 0, 1.5f);
68 }
69}

71void FTactixInfluenceRenderer::DrawAllChannels(const UWorld* World,
72 const Tactix::FTactixInfluenceMap<4>& Map,
73 float ZHeight,
74 float ChannelSpacing)
75{
76 if (!World) return;
77
78 static const TCHAR* ChannelNames[] = { TEXT("Danger"), TEXT("Control"), TEXT("Resource"), TEXT("Custom") };
79
80 const float GridWidth = Map.GetWidth() * Map.GetCellSize();
81 for (std::size_t C = 0; C < 4; ++C)
82 {
83 // Offset each channel along +Y so they don't overlap.
84 // We temporarily shift the origin by offsetting our sample point.
85 // Since the map's origin is fixed, we draw in the offset world space
86 // and label it.
87 const float YOffset = static_cast<float>(C) * (GridWidth + ChannelSpacing);
88
89 const int32 W = Map.GetWidth();
90 const int32 H = Map.GetHeight();
91 const float HalfCell = Map.GetCellSize() * 0.5f;
92
93 for (int32 y = 0; y < H; ++y)
94 for (int32 x = 0; x < W; ++x)
95 {
96 const Tactix::FTactixVec2 MapCenter = Map.CellCenter(x, y);
97 const float Value = Map.SampleNearest(C, MapCenter);
98 if (Value <= 0.01f) continue;
99
100 const FColor Color = HeatColor(FMath::Clamp(Value, 0.0f, 1.0f));
101 const FVector BoxCenter(MapCenter.X, MapCenter.Y + YOffset, ZHeight);
102 const FVector BoxExtent(HalfCell * 0.9f, HalfCell * 0.9f, 5.0f);
103
104 DrawDebugBox(World, BoxCenter, BoxExtent, FQuat::Identity, Color,
105 false, 0.0f, 0, 1.5f);
106 }
107
108 // Channel label
109 const Tactix::FTactixVec2 Origin = Map.GetOrigin();
110 const FVector LabelPos(Origin.X, Origin.Y + YOffset - 100.0f, ZHeight + 40.0f);
111 DrawDebugString(World, LabelPos, ChannelNames[C], nullptr, FColor::White,
112 0.0f, true, 1.2f);
113 }
114}
