41 bool Decompose(ITactixHTNTask* Task, FDecomposeCtx& C, uint32_t Depth)
42 {
43 if (Task == nullptr) return false;
44 if (Depth >= C.MaxDepth) return false;
45
46 if (Task->IsPrimitive())
47 {
48 auto* P = static_cast<ITactixHTNPrimitive*>(Task);
49 if (!P->IsApplicable(*C.State, *C.AgentCtx)) return false;
50 if (*C.PlanCount >= C.MaxPlan) return false;
51
52 C.Plan[(*C.PlanCount)++] = P;
53 C.State->Apply(P->GetEffects(*C.AgentCtx));
54 return true;
55 }
56
57 auto* Cmp = static_cast<ITactixHTNCompound*>(Task);
58 const uint32_t Methods = Cmp->GetMethodCount();
59
60 // Subtask buffer on the call stack; bounded by MaxSubs.
61 // Fixed at a sane upper limit to avoid VLAs (MSVC doesn't support them).
62 constexpr uint32_t kHardCap = 32;
63 ITactixHTNTask* Subs[kHardCap] = {nullptr};
64 const uint32_t Cap = C.MaxSubs < kHardCap ? C.MaxSubs : kHardCap;
65
66 for (uint32_t m = 0; m < Methods; ++m)
67 {
68 if (!Cmp->MethodApplies(m, *C.State, *C.AgentCtx)) continue;
69
70 const FTactixWorldState SnapState = *C.State;
71 const uint32_t SnapPlan = *C.PlanCount;
72
73 const uint32_t N = Cmp->GetMethodSubtasks(m, Subs, Cap);
74 if (N > Cap) continue;
75
76 bool bOk = true;
77 for (uint32_t i = 0; i < N; ++i)
78 {
79 if (!Decompose(Subs[i], C, Depth + 1)) { bOk = false; break; }
80 }
81 if (bOk) return true;
82
83 *C.State = SnapState;
84 *C.PlanCount = SnapPlan;
85 }
86 return false;
87 }
88 }

90 FTactixHTNPlan FTactixHTNPlanner::Plan(
91 ITactixHTNTask* Root,
92 const FTactixWorldState& InitialState,
93 const FTactixAgentContext& Ctx,
94 FTactixArena& Scratch,
95 FTactixHTNConfig Config)
96 {
97 FTactixHTNPlan Result{};
98
99 if (Root == nullptr || Config.MaxPlanLength == 0) return Result;
100
101 ITactixHTNPrimitive** Plan = Scratch.EmplaceArrayUninitialised<ITactixHTNPrimitive*>(Config.MaxPlanLength);
102 if (Plan == nullptr) return Result;
103
104 FTactixWorldState State = InitialState;
105 uint32_t Count = 0;
106
107 FDecomposeCtx C{
108 &State, Plan, &Count,
109 Config.MaxPlanLength, &Ctx,
110 Config.MaxRecursionDepth,
111 Config.MaxMethodSubtasks
112 };
113
114 if (!Decompose(Root, C, 0)) return Result;
115
116 Result.Steps = Plan;
117 Result.Count = Count;
118 Result.bValid = true;
119 return Result;
120 }
