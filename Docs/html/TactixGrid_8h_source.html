173 int32_t GetNeighbours(int32_t X, int32_t Y, std::size_t OutIndices[8]) const
174 {
175 static constexpr int32_t DX[8] = { -1, 0, 1, -1, 1, -1, 0, 1 };
176 static constexpr int32_t DY[8] = { -1, -1, -1, 0, 0, 1, 1, 1 };
177 int32_t N = 0;
178 for (int32_t i = 0; i < 8; ++i)
179 {
180 const int32_t NX = X + DX[i];
181 const int32_t NY = Y + DY[i];
182 if (InBounds(NX, NY))
183 {
184 OutIndices[N++] = Index(NX, NY);
185 }
186 }
187 return N;
188 }
