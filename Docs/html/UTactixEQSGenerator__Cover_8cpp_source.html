29
30 Tactix::FTactixCoverSystem<512>& CoverSys = TactixSub->GetCoverSystem();
31
32 TArray<FNavLocation> CoverLocations;

34 const Tactix::FTactixCoverPoint& P)

37 CoverLocations.Add(FNavLocation(FVector(P.Position.X, P.Position.Y, P.Position.Z)));
38 });
39
40 QueryInstance.ReserveItemData(CoverLocations.Num());
41 for (const FNavLocation& Loc : CoverLocations)
42 {
43 QueryInstance.AddItemData<UEnvQueryItemType_Point>(Loc);
44 }
45}
