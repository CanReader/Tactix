33void FTactixGameplayDebuggerCategory::CollectData(APlayerController* OwnerPC, AActor* DebugActor)
34{
35 const APawn* Pawn = Cast<APawn>(DebugActor);
36 if (!Pawn)
37 {
38 AddTextLine(FString(TEXT("{red}Selected actor is not a pawn")));
39 return;
40 }
41
42 const ATactixAIController* Controller =
43 Cast<ATactixAIController>(Pawn->GetController());
44 if (!Controller)
45 {
46 AddTextLine(FString(TEXT("{red}Pawn has no ATactixAIController")));
47 return;
48 }
49
50 const UTactixAgentComponent* Agent = Controller->GetTactixAgent();
51 if (!Agent)
52 {
53 AddTextLine(FString(TEXT("{red}No UTactixAgentComponent found")));
54 return;
55 }
56
57 // Vitals
58 const float HP = (Agent->MaxHealth > 0.0f) ? Agent->CurrentHealth / Agent->MaxHealth : 0.0f;
59 const float Amm = (Agent->MaxAmmo > 0.0f) ? Agent->CurrentAmmo / Agent->MaxAmmo : 0.0f;
60 const float Sta = (Agent->MaxStamina > 0.0f) ? Agent->CurrentStamina / Agent->MaxStamina : 0.0f;
61
62 const auto PctColor = [](float R) -> const TCHAR*
63 {
64 return (R > 0.6f) ? TEXT("{green}") : (R > 0.3f) ? TEXT("{yellow}") : TEXT("{red}");
65 };
66
67 AddTextLine(FString::Printf(TEXT("{white}HP %s%.0f%% {white}Ammo %s%.0f%% {white}Sta %s%.0f%%"),
68 PctColor(HP), HP * 100.0f,
69 PctColor(Amm), Amm * 100.0f,
70 PctColor(Sta), Sta * 100.0f));
71
72 // State flags
73 FString State;
74 if (Agent->bInCover) State += TEXT("[Cover] ");
75 if (Agent->bReloading) State += TEXT("[Reload] ");
76 if (Agent->bHasThreat)
77 {
78 const float Dist = (float)(Pawn->GetActorLocation() - Agent->ThreatLocation).Size();
79 State += FString::Printf(TEXT("[Threat %.0f uu]"), Dist);
80 }
81 AddTextLine(FString::Printf(TEXT("{white}State: {yellow}%s"), State.IsEmpty() ? TEXT("–") : *State));
82
83 // Active plan
84 FString PlanStr;
85 if (Controller->bGOAPPlanActive && Controller->ActiveGOAPPlan.bValid)
86 {
87 PlanStr = FString::Printf(TEXT("GOAP %u/%u"),
88 Controller->GOAPPlanStep + 1, Controller->ActiveGOAPPlan.Count);
89 }
90 else if (Controller->bHTNPlanActive && Controller->ActiveHTNPlan.bValid)
91 {
92 PlanStr = FString::Printf(TEXT("HTN %u/%u"),
93 Controller->HTNPlanStep + 1, Controller->ActiveHTNPlan.Count);
94 }
95 else
96 {
97 PlanStr = TEXT("idle");
98 }
99 AddTextLine(FString::Printf(TEXT("{white}Plan: {cyan}%s"), *PlanStr));
100
101 // Last decision from the recorder
102 const auto History = FTactixDecisionRecorder::Get().GetHistory(Agent->GetHandle(), 1);
103 if (!History.IsEmpty())
104 {
105 const FTactixDecisionRecord& Last = History[0];
106 AddTextLine(FString::Printf(TEXT("{white}Action: {orange}%s {white}score={orange}%.3f"),
107 *Last.ActionName.ToString(), Last.Score));
108 }
109 else
110 {
111 AddTextLine(FString(TEXT("{white}Action: {grey}–")));
112 }
113}
