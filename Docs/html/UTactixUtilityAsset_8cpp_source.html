19float UTactixUtilityAsset::GetRawInput(const FTactixConsiderationConfig& Cfg,
20 const FTactixAgentContext& Ctx)
21{
22 switch (Cfg.InputType)
23 {
24 case ETactixInputType::HealthRatio:
25 return Ctx.HealthRatio;
26
27 case ETactixInputType::AmmoRatio:
28 return Ctx.AmmoRatio;
29
30 case ETactixInputType::StaminaRatio:
31 return Ctx.StaminaRatio;
32
33 case ETactixInputType::ThreatDistance:
34 // Negative ThreatDistance means no known threat → input is 0.
35 if (Ctx.ThreatDistance < 0.0f) return 0.0f;
36 // Map [0, Cap] → [1, 0]: close threat = high urgency.
37 return 1.0f - Clamp(Ctx.ThreatDistance / FMath::Max(Cfg.ThreatDistanceCap, 1.0f), 0.0f, 1.0f);
38
39 case ETactixInputType::Speed:
40 // Normalise against 600 uu/s (roughly a jog in UE defaults).
41 return Clamp(Ctx.Velocity.Length() / 600.0f, 0.0f, 1.0f);
42
43 case ETactixInputType::InCover:
44 return Ctx.bInCover ? 1.0f : 0.0f;
45
46 default:
47 return 0.0f;
48 }
49}

76float UTactixUtilityAsset::ScoreAction(int32 Index, const FTactixAgentContext& Ctx) const
77{
78 if (!Actions.IsValidIndex(Index)) return 0.0f;
79 const FTactixActionConfig& Action = Actions[Index];
80 if (Action.Considerations.IsEmpty()) return 0.0f;
81
82 const int32 N = Action.Considerations.Num();
83 const float ModFactor = 1.0f - (1.0f / static_cast<float>(N));
84 float Product = 1.0f;
85
86 for (const FTactixConsiderationConfig& Cfg : Action.Considerations)
87 {
88 const float Raw = GetRawInput(Cfg, Ctx);
89 if (Raw <= 0.0f) return 0.0f;
90
91 // Apply the response curve, clamp to [0, 1].
92 float X = Clamp(MakeCurve(Cfg).Evaluate(Raw), 0.0f, 1.0f);
93 if (X <= 0.0f) return 0.0f;
94
95 // Mike Lewis compensation so extra considerations don't kill the score.
96 const float MakeUp = (1.0f - X) * ModFactor;
97 X += MakeUp * X;
98
99 Product *= X;
100 }
101
102 return Product;
103}

105int32 UTactixUtilityAsset::PickBestAction(const FTactixAgentContext& Ctx,
106 TArray<float>* OutScores) const
107{
108 int32 BestIdx = INDEX_NONE;
109 float BestScore = 0.0f;
110
111 for (int32 i = 0; i < Actions.Num(); ++i)
112 {
113 const float S = ScoreAction(i, Ctx);
114 if (OutScores) OutScores->Add(S);
115 if (S > BestScore)
116 {
117 BestScore = S;
118 BestIdx = i;
119 }
120 }
121
122 return BestIdx;
123}
