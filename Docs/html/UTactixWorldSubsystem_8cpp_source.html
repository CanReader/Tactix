24void UTactixWorldSubsystem::ReserveCover(const AActor* Reserver, const AActor* CoverActor, float Lifetime)
25{
26 if (!Reserver || !CoverActor) return;
27 const UWorld* W = GetWorld();
28 if (!W) return;
29
30 FCoverReservation R;
31 R.Reserver = Reserver;
32 R.ExpiresAt = W->GetTimeSeconds() + Lifetime;
33 CoverReservations.Add(CoverActor, R);
34}

36bool UTactixWorldSubsystem::IsCoverReservedByOther(const AActor* Reserver, const AActor* CoverActor) const
37{
38 if (!CoverActor) return false;
39 const UWorld* W = GetWorld();
40 if (!W) return false;
41
42 const FCoverReservation* Existing = CoverReservations.Find(CoverActor);
43 if (!Existing) return false;
44 if (Existing->ExpiresAt <= W->GetTimeSeconds()) return false;
45 const AActor* HeldBy = Existing->Reserver.Get();
46 return HeldBy != nullptr && HeldBy != Reserver;
47}
