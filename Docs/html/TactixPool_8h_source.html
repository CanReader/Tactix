108 TACTIX_NODISCARD HandleType Alloc(Args&&... InArgs)
109 {
110 if (FreeHead == kEndOfList)
111 {
112 return HandleType::Invalid();
113 }
114 const uint16_t Index = FreeHead;
115 FreeHead = Slots[Index].NextFree;
116
117 ::new (static_cast<void*>(Slots[Index].Storage)) T(std::forward<Args>(InArgs)...);
118 Slots[Index].bAlive = true;
119 ++AliveCount;
120
121 return HandleType{Index, Slots[Index].Generation};
122 }

136 bool Free(HandleType InHandle)
137 {
138 if (!InHandle.IsValid()) return false;
139 const uint32_t Idx32 = InHandle.GetIndex();
140 if (Idx32 >= Capacity) return false;
141 const uint16_t Index = static_cast<uint16_t>(Idx32);
142
143 if (!Slots[Index].bAlive) return false;
144 if (Slots[Index].Generation != InHandle.GetGeneration()) return false;
145
146 if constexpr (!std::is_trivially_destructible_v<T>)
147 {
148 AtSlot(Index)->~T();
149 }
150 Slots[Index].bAlive = false;
151
152 // Bump generation, skipping zero so the "invalid handle" sentinel stays unique.
153 uint16_t NextGen = static_cast<uint16_t>(Slots[Index].Generation + 1u);
154 if (NextGen == 0) NextGen = 1;
155 Slots[Index].Generation = NextGen;
156
157 Slots[Index].NextFree = FreeHead;
158 FreeHead = Index;
159 --AliveCount;
160 return true;
161 }
