88 static float ScoreConsiderations(const ITactixConsideration* const* Considerations,
89 std::size_t Count, const FTactixAgentContext& Ctx)
90 {
91 if (Count == 0 || Considerations == nullptr) return 0.0f;
92
93 const float ModFactor = 1.0f - (1.0f / static_cast<float>(Count));
94 float Product = 1.0f;
95
96 for (std::size_t i = 0; i < Count; ++i)
97 {
98 const ITactixConsideration* C = Considerations[i];
99 if (C == nullptr) return 0.0f;
100
101 float X = C->Evaluate(Ctx);
102 if (X <= 0.0f) return 0.0f;
103 if (X > 1.0f) X = 1.0f;
104
105 const float MakeUp = (1.0f - X) * ModFactor;
106 X = X + (MakeUp * X);
107
108 Product *= X;
109 }
110
111 return Product;
