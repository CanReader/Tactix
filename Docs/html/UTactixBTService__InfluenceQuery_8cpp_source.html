33 UTactixWorldSubsystem* TactixSub = World->GetSubsystem<UTactixWorldSubsystem>();
34 if (!TactixSub) return;
35
36 const ATactixAIController* Controller = Cast<ATactixAIController>(OwnerComp.GetAIOwner());
37 if (!Controller) return;
38

41
42 const Tactix::FTactixAgentContext Ctx = Agent->BuildContext();
43 const Tactix::FTactixVec2 XY{ Ctx.Position.X, Ctx.Position.Y };
44
45 const Tactix::FTactixInfluenceMap<4>& Map = TactixSub->GetInfluenceMap();
46 const std::size_t Ch = static_cast<std::size_t>(FMath::Clamp(ChannelIndex, 0, 3));
47
48 const float Value = bBilinear ? Map.SampleBilinear(Ch, XY) : Map.SampleNearest(Ch, XY);
49
50 if (UBlackboardComponent* BB = OwnerComp.GetBlackboardComponent())

52 if (InfluenceValueKey.IsSet())
53 {
54 BB->SetValueAsFloat(InfluenceValueKey.SelectedKeyName, Value);

56 }
