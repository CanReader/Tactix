76 void Build(ETactixFormationKind Kind, uint32_t Count, float Spacing)
77 {
78 this->Kind = Kind;
79 this->Spacing = Spacing > 0.0f ? Spacing : 100.0f;
80 this->Count = Count > MaxSlots ? static_cast<uint32_t>(MaxSlots) : Count;
81
82 switch (Kind)
83 {
84 case ETactixFormationKind::Line: BuildLine(); break;
85 case ETactixFormationKind::Column: BuildColumn(); break;
86 case ETactixFormationKind::Wedge: BuildWedge(); break;
87 case ETactixFormationKind::Box: BuildBox(); break;
88 case ETactixFormationKind::Diamond: BuildDiamond(); break;
89 case ETactixFormationKind::Custom: break; // keep current slots
90 }
91 }

130 FTactixVec3 GetSlotWorldPosition(uint32_t SlotIdx,
131 FTactixVec3 LeaderPos,
132 FTactixVec3 LeaderForward) const
133 {
134 if (SlotIdx >= Count) return LeaderPos;
135
136 FTactixVec3 F{ LeaderForward.X, LeaderForward.Y, 0.0f };
137 if (F.LengthSquared() < 1e-6f) { F = { 1.0f, 0.0f, 0.0f }; }
138 else { F = F.Normalized(); }
139
140 // Right = Up x Forward (UE left-handed convention).
141 const FTactixVec3 R{ -F.Y, F.X, 0.0f };
142 const FTactixVec3 U{ 0.0f, 0.0f, 1.0f };
143
144 const FTactixVec3& L = Slots[SlotIdx].LocalOffset;
145 return LeaderPos + (F * L.X) + (R * L.Y) + (U * L.Z);
146 }

155 void BuildLine()
156 {
157 Slots[0].LocalOffset = {};
158 for (uint32_t i = 1; i < Count; ++i)
159 {
160 const int32_t step = static_cast<int32_t>((i + 1) / 2);
161 const float sign = (i & 1u) ? 1.0f : -1.0f;
162 Slots[i].LocalOffset = { 0.0f, sign * static_cast<float>(step) * Spacing, 0.0f };
163 }
164 }
165
167 void BuildColumn()
168 {
169 for (uint32_t i = 0; i < Count; ++i)
170 {
171 Slots[i].LocalOffset = { -static_cast<float>(i) * Spacing, 0.0f, 0.0f };
172 }
173 }
174
176 void BuildWedge()
177 {
178 Slots[0].LocalOffset = {};
179 for (uint32_t i = 1; i < Count; ++i)
180 {
181 const int32_t step = static_cast<int32_t>((i + 1) / 2);
182 const float ySig = (i & 1u) ? -1.0f : 1.0f; // odd=right, even=left
183 Slots[i].LocalOffset = {
184 -static_cast<float>(step) * Spacing,
185 ySig * static_cast<float>(step) * Spacing,
186 0.0f
187 };
188 }
189 }
190
192 void BuildBox()
193 {
194 const FTactixVec3 Pattern[4] = {
195 { 0.0f, 0.0f, 0.0f },
196 { 0.0f, Spacing, 0.0f },
197 {-Spacing, 0.0f, 0.0f },
198 {-Spacing, Spacing, 0.0f },
199 };
200 for (uint32_t i = 0; i < Count; ++i)
201 {
202 const uint32_t tier = i / 4u;
203 const uint32_t p = i % 4u;
204 FTactixVec3 off = Pattern[p];
205 off.X -= static_cast<float>(tier) * Spacing * 2.0f;
206 Slots[i].LocalOffset = off;
207 }
208 }
209
211 void BuildDiamond()
212 {
213 const FTactixVec3 Pattern[4] = {
214 { Spacing, 0.0f, 0.0f },
215 { 0.0f, Spacing,0.0f },
216 { -Spacing, 0.0f, 0.0f },
217 { 0.0f, -Spacing,0.0f },
218 };
219 for (uint32_t i = 0; i < Count; ++i)
220 {
221 Slots[i].LocalOffset = Pattern[i % 4u];
222 }
223 }
224 };
