153 static float ScoreOne(const FTactixCoverPoint& P, const FQuery& Q)
154 {
155 if (P.IsClaimed() && Q.bExcludeClaimed && P.ClaimedBy != Q.Querier) return 0.0f;
156
157 const FTactixVec3 ToThreat = Q.ThreatPosition - P.Position;
158 const float ThreatDistSq = ToThreat.LengthSquared();
159 if (ThreatDistSq < 1e-4f) return 0.0f;
160
161 const float ThreatDist = std::sqrt(ThreatDistSq);
162 const FTactixVec3 ThreatDir = ToThreat / ThreatDist;
163
164 const float Block = -ThreatDir.Dot(P.Normal);
165 if (Block < Q.MinBlockDot) return 0.0f;
166
167 const FTactixVec3 ToPoint = P.Position - Q.FromPosition;
168 const float Range = ToPoint.Length();
169 if (Range > Q.MaxRange) return 0.0f;
170
171 const float RangeFactor = (Q.MaxRange > 0.0f) ? (1.0f - (Range / Q.MaxRange)) : 1.0f;
172 const float BlockNorm = 0.5f * (Block + 1.0f);
173 const float HeightBias = (P.Height == ETactixCoverHeight::High) ? 1.0f : 0.9f;
174
175 return BlockNorm * RangeFactor * HeightBias;
176 }
