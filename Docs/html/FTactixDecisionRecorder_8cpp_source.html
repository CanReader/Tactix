29void FTactixDecisionRecorder::Record(const FTactixDecisionRecord& Entry)
30{
31 FTactixDecisionRecord Stamped = Entry;
32 Stamped.WallTimeSeconds = FPlatformTime::Seconds();
33
34 if (Ring.Num() < kRingSize)
35 {
36 Ring.Add(Stamped);
37 }
38 else
39 {
40 Ring[Head] = Stamped;
41 }
42 Head = (Head + 1) % kRingSize;
43 Count = FMath::Min(Count + 1, kRingSize);
44}

48TArray<FTactixDecisionRecord> FTactixDecisionRecorder::GetHistory(
49 Tactix::FTactixHandle<Tactix::ITactixAgent> Agent, int32 MaxEntries) const
50{
51 TArray<FTactixDecisionRecord> Result;
52 // Walk backwards from the most recent entry.
53 const int32 Total = Ring.Num();
54 for (int32 i = 0; i < Total && Result.Num() < MaxEntries; ++i)
55 {
56 // Most recent first: index = (Head - 1 - i + kRingSize) % kRingSize
57 const int32 Idx = ((Head - 1 - i) % kRingSize + kRingSize) % kRingSize;
58 if (!Ring.IsValidIndex(Idx)) continue;
59 if (Ring[Idx].AgentHandle == Agent)
60 {
61 Result.Add(Ring[Idx]);
62 }
63 }
64 return Result;
65}

67TArray<FTactixDecisionRecord> FTactixDecisionRecorder::GetAll() const
68{
69 TArray<FTactixDecisionRecord> Result;
70 const int32 Total = Ring.Num();
71 // Return in chronological order.
72 for (int32 i = 0; i < Total; ++i)
73 {
74 const int32 Idx = (Head - Total + i + kRingSize * 2) % kRingSize;
75 if (Ring.IsValidIndex(Idx))
76 {
77 Result.Add(Ring[Idx]);
78 }
79 }
80 return Result;
81}
