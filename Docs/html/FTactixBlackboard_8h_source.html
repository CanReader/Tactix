46 constexpr uint32_t HashFNV1a(const char* Text) noexcept
47 {
48 uint32_t H = 0x811c9dc5u;
49 while (*Text != '\0')
50 {
51 H ^= static_cast<uint32_t>(static_cast<uint8_t>(*Text));
52 H *= 0x01000193u;
53 ++Text;
54 }
55 return H == 0u ? 1u : H;
56 }

281 bool Remove(FTactixBBKey Key) override
282 {
283 std::size_t Hole = Find(Key);
284 if (Hole == Capacity) return false;
285
286 Keys[Hole] = {};
287 Values[Hole] = {};
288 --Size;
289
290 // Open-addressed backshift: walk forward and move any entry whose
291 // "home" slot sits cyclically in [Home, J) across the hole — if
292 // yes, dragging it back to the hole preserves probe invariants.
293 std::size_t J = (Hole + 1) & Mask;
294 while (Keys[J].IsValid())
295 {
296 const std::size_t Home = Keys[J].Hash & Mask;
297 const bool InRange = (Home <= J) ? (Hole >= Home && Hole < J)
298 : (Hole >= Home || Hole < J);
299 if (!InRange) break;
300
301 Keys[Hole] = Keys[J];
302 Values[Hole] = Values[J];
303 Keys[J] = {};
304 Values[J] = {};
305 Hole = J;
306 J = (J + 1) & Mask;
307 }
308 return true;
309 }

311 void Clear() override
312 {
313 for (std::size_t i = 0; i < Capacity; ++i) { Keys[i] = {}; Values[i] = {}; }
314 Size = 0;
315 }

329 std::size_t Find(FTactixBBKey Key) const
330 {
331 if (!Key.IsValid()) return Capacity;
332 std::size_t I = Key.Hash & Mask;
333 for (std::size_t Probes = 0; Probes < Capacity; ++Probes)
334 {
335 if (!Keys[I].IsValid()) return Capacity; // empty slot = not found
336 if (Keys[I].Hash == Key.Hash) return I;
337 I = (I + 1) & Mask;
338 }
339 return Capacity;
340 }
341
347 std::size_t FindOrInsert(FTactixBBKey Key)
348 {
349 if (!Key.IsValid()) return Capacity;
350 std::size_t I = Key.Hash & Mask;
351 for (std::size_t Probes = 0; Probes < Capacity; ++Probes)
352 {
353 if (!Keys[I].IsValid())
354 {
355 Keys[I] = Key;
356 ++Size;
357 return I;
358 }
359 if (Keys[I].Hash == Key.Hash) return I;
360 I = (I + 1) & Mask;
361 }
362 return Capacity; // table full
363 }
