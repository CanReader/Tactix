31 void FTactixAgentScheduler::Grow(std::size_t NewCapacity)
32 {
33 if (NewCapacity <= Cap) return;
34 auto* NewHeap = new FTactixAgentScheduleEntry[NewCapacity];
35 if (Count > 0 && Heap != nullptr)
36 {
37 for (std::size_t i = 0; i < Count; ++i) NewHeap[i] = Heap[i];

39 delete[] Heap;
40 Heap = NewHeap;

61 bool FTactixAgentScheduler::Register(const FTactixAgentScheduleEntry& Entry)
62 {
63 if (!Entry.Agent.IsValid()) return false;
64
65 const std::size_t Existing = FindIndex(Entry.Agent);
66 if (Existing != Cap)
67 {
68 // Replace-in-place and re-sift both directions (either may be required).
69 Heap[Existing] = Entry;
70 SiftUp(Existing);
71 SiftDown(Existing);
72 return true;
73 }
74
75 if (Count == Cap) Grow(Cap * 2);
76 Heap[Count] = Entry;
77 SiftUp(Count);
78 ++Count;
79 return true;
80 }

96 void FTactixAgentScheduler::SiftUp(std::size_t Index)
97 {
98 while (Index > 0)
99 {
100 const std::size_t Parent = (Index - 1) / 2;
101 if (Less(Heap[Index], Heap[Parent]))
102 {
103 std::swap(Heap[Index], Heap[Parent]);
104 Index = Parent;
105 }
106 else
107 {
108 break;
109 }
110 }
111 }

113 void FTactixAgentScheduler::SiftDown(std::size_t Index)
114 {
115 while (true)
116 {
117 const std::size_t Left = Index * 2 + 1;
118 const std::size_t Right = Index * 2 + 2;
119 std::size_t Smallest = Index;
120
121 if (Left < Count && Less(Heap[Left], Heap[Smallest])) Smallest = Left;
122 if (Right < Count && Less(Heap[Right], Heap[Smallest])) Smallest = Right;
123 if (Smallest == Index) return;
124
125 std::swap(Heap[Smallest], Heap[Index]);
126 Index = Smallest;
127 }
128 }

130 std::size_t FTactixAgentScheduler::Collect(double NowSeconds,
131 uint32_t FrameBudgetUs,
132 FTactixHandle<ITactixAgent>* OutAgents,
133 std::size_t OutCapacity)
134 {
135 if (OutAgents == nullptr || OutCapacity == 0) return 0;
136
137 uint32_t SpentUs = 0;
138 std::size_t NumOut = 0;
139
140 while (Count > 0 && NumOut < OutCapacity)
141 {
142 FTactixAgentScheduleEntry& Top = Heap[0];
143 if (Top.NextTickTime > NowSeconds) break;
144
145 // Budget check: if adding this agent's expected cost would blow the
146 // frame budget, stop — but always allow the very first agent to
147 // run, otherwise an expensive agent could starve forever.
148 const uint32_t ExpectedUs = Top.LastSpentUs == 0 ? Top.BudgetUs : Top.LastSpentUs;
149 if (NumOut > 0 && SpentUs + ExpectedUs > FrameBudgetUs) break;
150
151 OutAgents[NumOut++] = Top.Agent;
152 SpentUs += ExpectedUs;
153
154 // Re-schedule at now + interval so we don't double-tick.
155 Top.NextTickTime = NowSeconds + (double)Top.TickIntervalMs / 1000.0;
156 SiftDown(0);
157 }
158 return NumOut;
159 }
