103 void Stamp(std::size_t Channel, FTactixVec2 World, float Radius, float Value)
104 {
105 if (Channel >= ChannelCount || Data == nullptr || Radius <= 0.0f) return;
106
107 int32_t CX, CY;
108 WorldToCell(World, CX, CY);
109 const int32_t CellRadius = static_cast<int32_t>(std::ceil(Radius / CellSize));
110 const int32_t X0 = Max(0, CX - CellRadius);
111 const int32_t X1 = Min(Width - 1, CX + CellRadius);
112 const int32_t Y0 = Max(0, CY - CellRadius);
113 const int32_t Y1 = Min(Height - 1, CY + CellRadius);
114
115 float* Ch = ChannelData(Channel);
116 const float InvR = 1.0f / Radius;
117
118 for (int32_t y = Y0; y <= Y1; ++y)
119 for (int32_t x = X0; x <= X1; ++x)
120 {
121 const FTactixVec2 C = CellCenter(x, y);
122 const float DX = C.X - World.X;
123 const float DY = C.Y - World.Y;
124 const float D = std::sqrt(DX * DX + DY * DY);
125 if (D >= Radius) continue;
126 Ch[Idx(x, y)] += Value * (1.0f - D * InvR);
127 }
128 }

178 float SampleBilinear(std::size_t Channel, FTactixVec2 World) const
179 {
180 if (Channel >= ChannelCount || Data == nullptr) return 0.0f;
181
182 const float U = (World.X - Origin.X) / CellSize - 0.5f;
183 const float V = (World.Y - Origin.Y) / CellSize - 0.5f;
184 const int32_t X0 = static_cast<int32_t>(std::floor(U));
185 const int32_t Y0 = static_cast<int32_t>(std::floor(V));
186 const float FX = U - static_cast<float>(X0);
187 const float FY = V - static_cast<float>(Y0);
188
189 const float* Ch = ChannelData(Channel);
190 auto At = [&](int32_t X, int32_t Y) -> float
191 {
192 if (X < 0 || Y < 0 || X >= Width || Y >= Height) return 0.0f;
193 return Ch[Idx(X, Y)];
194 };
195
196 const float A = At(X0, Y0 );
197 const float B = At(X0 + 1, Y0 );
198 const float C = At(X0, Y0 + 1);
199 const float D = At(X0 + 1, Y0 + 1);
200 const float Top = A + (B - A) * FX;
201 const float Bottom = C + (D - C) * FX;
202 return Top + (Bottom - Top) * FY;
203 }
