29void ATactixAIController::OnPossess(APawn* InPawn)
30{
31 Super::OnPossess(InPawn);
32
33 if (UTactixAgentComponent* Agent = GetTactixAgent())
34 {
35 // Use the lower 16 bits of the pawn's unique ID as the handle index.
36 // Generation 1 means "first occupant of this slot" — never zero.
37 const uint32 Idx = InPawn->GetUniqueID() & 0xFFFFu;
38 Agent->SetHandle(Tactix::FTactixHandle<Tactix::ITactixAgent>{Idx, 1u});
39 }
40}
