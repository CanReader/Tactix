20 float FTactixCurve::Evaluate(float X) const noexcept
21 {
22 switch (Type)
23 {
24 case ETactixCurveType::Linear:
25 {
26 const float Y = Slope * (X - Shift);
27 return Saturate(Y);
28 }
29
30 case ETactixCurveType::Quadratic:
31 {
32 // std::pow of a negative base with a non-integer exponent is NaN;
33 // we intentionally preserve the sign by raising |arg|^Exp and copying the sign back.
34 const float Arg = X - Shift;
35 const float Magnitude = std::pow(std::fabs(Arg), Exp);
36 const float Signed = Arg < 0.0f ? -Magnitude : Magnitude;
37 return Saturate(Slope * Signed);
38 }
39
40 case ETactixCurveType::Logistic:
41 {
42 // Classic sigmoid; already in (0, 1) but still saturate in case
43 // of FP drift at extremes of Slope.
44 const float Z = -Slope * (X - Shift);
45 const float Y = 1.0f / (1.0f + std::exp(Z));
46 return Saturate(Y);
47 }
48
49 case ETactixCurveType::Exponential:
50 {
51 const float Y = Slope * (std::exp(Exp * (X - Shift)) - 1.0f);
52 return Saturate(Y);
53 }
54
55 case ETactixCurveType::Custom:
56 {
57 if (CustomFn == nullptr) return 0.0f;
58 return Saturate(CustomFn(X));
59 }
60 }
61 return 0.0f;
62 }
