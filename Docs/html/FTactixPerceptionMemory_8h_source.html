66 void Decay(double NowSeconds, float DefaultHalfLife = 5.0f, float Floor = 0.05f)
67 {
68 const float SafeHalfLife = DefaultHalfLife > 0.0f ? DefaultHalfLife : 1.0f;
69 for (std::size_t i = 0; i < Size; ++i)
70 {
71 FTactixStimulus& S = PeekMutable(i);
72 if (S.Intensity <= 0.0f) continue;
73 const float Age = static_cast<float>(NowSeconds - S.Timestamp);
74 if (Age <= 0.0f) continue;
75 const float Decayed = S.Intensity * std::exp2(-Age / SafeHalfLife);
76 S.Intensity = Decayed < Floor ? 0.0f : Decayed;
77 }
78 }
