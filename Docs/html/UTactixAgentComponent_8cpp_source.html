20FTactixAgentContext UTactixAgentComponent::BuildContext() const
21{
22 FTactixAgentContext Ctx{};
23 Ctx.AgentHandle = Handle;
24
25 const AActor* Owner = GetOwner();
26 if (Owner)
27 {
28 const FVector Pos = Owner->GetActorLocation();
29 Ctx.Position = { (float)Pos.X, (float)Pos.Y, (float)Pos.Z };
30
31 const FVector Fwd = Owner->GetActorForwardVector();
32 Ctx.Forward = { (float)Fwd.X, (float)Fwd.Y, (float)Fwd.Z };
33
34 // Try to get velocity from the root primitive's physics state.
35 if (UPrimitiveComponent* Root =
36 Cast<UPrimitiveComponent>(Owner->GetRootComponent()))
37 {
38 const FVector Vel = Root->GetPhysicsLinearVelocity();
39 Ctx.Velocity = { (float)Vel.X, (float)Vel.Y, (float)Vel.Z };
40 }
41 }
42
43 // Normalise vitals; guards against zero max values.
44 Ctx.HealthRatio = (MaxHealth > 0.0f) ? FMath::Clamp(CurrentHealth / MaxHealth, 0.0f, 1.0f) : 0.0f;
45 Ctx.AmmoRatio = (MaxAmmo > 0.0f) ? FMath::Clamp(CurrentAmmo / MaxAmmo, 0.0f, 1.0f) : 0.0f;
46 Ctx.StaminaRatio = (MaxStamina > 0.0f) ? FMath::Clamp(CurrentStamina / MaxStamina, 0.0f, 1.0f) : 0.0f;
47
48 Ctx.bInCover = bInCover;
49 Ctx.bReloading = bReloading;
50
51 if (bHasThreat && Owner)
52 {
53 Ctx.ThreatLocation = { (float)ThreatLocation.X, (float)ThreatLocation.Y, (float)ThreatLocation.Z };
54 Ctx.ThreatDistance = (float)(Owner->GetActorLocation() - ThreatLocation).Size();
55 }
56
57 if (const UWorld* W = GetWorld())
58 {
59 Ctx.TimeSeconds = W->GetTimeSeconds();
60 }
61
62 return Ctx;
63}
