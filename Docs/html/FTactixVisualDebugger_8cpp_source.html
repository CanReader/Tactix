22FColor FTactixVisualDebugger::RatioColor(float R)
23{
24 R = FMath::Clamp(R, 0.0f, 1.0f);
25 if (R < 0.5f)
26 {
27 // Red → Yellow
28 return FColor(255, static_cast<uint8>(R * 2.0f * 255.0f), 0);
29 }
30 else
31 {
32 // Yellow → Green
33 return FColor(static_cast<uint8>((1.0f - R) * 2.0f * 255.0f), 255, 0);
34 }
35}

39void FTactixVisualDebugger::DrawAgent(const UWorld* World,
40 const ATactixAIController* Controller,
41 const FTactixDebugSettings& Settings)
42{
43 if (!World || !Controller) return;
44
45 const UTactixAgentComponent* Agent = Controller->GetTactixAgent();
46 if (!Agent) return;
47
48 const APawn* Pawn = Controller->GetPawn();
49 if (!Pawn) return;
50
51 const FVector Base = Pawn->GetActorLocation();
52 FVector TextPos = Base + FVector(0.0f, 0.0f, Settings.ZOffset);
53
54 // -- Vital bars --
55 if (Settings.bShowVitalBars)
56 {
57 const float HP = (Agent->MaxHealth > 0.0f) ? (Agent->CurrentHealth / Agent->MaxHealth) : 0.0f;
58 const float Amm = (Agent->MaxAmmo > 0.0f) ? (Agent->CurrentAmmo / Agent->MaxAmmo) : 0.0f;
59 const float Sta = (Agent->MaxStamina > 0.0f) ? (Agent->CurrentStamina / Agent->MaxStamina) : 0.0f;
60
61 DrawDebugString(World, TextPos,
62 FString::Printf(TEXT("HP %.0f%% Ammo %.0f%% Sta %.0f%%"),
63 HP * 100.0f, Amm * 100.0f, Sta * 100.0f),
64 nullptr, RatioColor(HP), 0.0f, true, 1.1f);
65 TextPos.Z += 22.0f;
66 }
67
68 // -- State flags --
69 if (Settings.bShowStateLine)
70 {
71 FString State;
72 if (Agent->bInCover) State += TEXT("[COVER] ");
73 if (Agent->bReloading) State += TEXT("[RELOAD] ");
74 if (Agent->bHasThreat)
75 {
76 const float Dist = (float)(Pawn->GetActorLocation() - Agent->ThreatLocation).Size();
77 State += FString::Printf(TEXT("[THREAT %.0fuu]"), Dist);
78 }
79 if (!State.IsEmpty())
80 {
81 DrawDebugString(World, TextPos, State, nullptr, FColor::Yellow, 0.0f, true, 1.0f);
82 TextPos.Z += 22.0f;
83 }
84 }
85
86 // -- Threat line --
87 if (Settings.bShowThreatLine && Agent->bHasThreat)
88 {
89 DrawDebugLine(World, Base, Agent->ThreatLocation, FColor::Red, false, 0.0f, 0, 1.5f);
90 }
91
92 // -- GOAP plan steps --
93 if (Settings.bShowPlanSteps)
94 {
95 DrawGOAPPlan(World, Base, Controller->ActiveGOAPPlan,
96 Controller->GOAPPlanStep, Settings.ZOffset + 22.0f * 2.0f);
97 }
98}

134void FTactixVisualDebugger::DrawUtilityScores(const UWorld* World,
135 const FVector& AgentPos,
136 const TArray<TPair<FName, float>>& ActionScores,
137 float ZOffset)
138{
139 if (!World || ActionScores.IsEmpty()) return;
140
141 FVector TextPos = AgentPos + FVector(0.0f, 0.0f, ZOffset);
142 for (const TPair<FName, float>& Pair : ActionScores)
143 {
144 const FColor C = RatioColor(Pair.Value);
145 DrawDebugString(World, TextPos,
146 FString::Printf(TEXT("%s %.2f"), *Pair.Key.ToString(), Pair.Value),
147 nullptr, C, 0.0f, true, 0.9f);
148 TextPos.Z += 18.0f;
149 }
150}
