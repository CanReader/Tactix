26bool UTactixBTTask_ExecutePlan::StepPlan(UBehaviorTreeComponent& OwnerComp) const
27{
28 ATactixAIController* Controller = Cast<ATactixAIController>(OwnerComp.GetAIOwner());
29 if (!Controller) return true; // signal "done" so the task fails out
30
31 UTactixAgentComponent* Agent = Controller->GetTactixAgent();
32 if (!Agent) return true;
33
34 Tactix::FTactixAgentContext Ctx = Agent->BuildContext();
35
36 if (PlanType == ETactixPlanType::GOAP)
37 {
38 if (!Controller->ActiveGOAPPlan.bValid) return true;
39 const uint32 Step = Controller->GOAPPlanStep;
40 if (Step >= Controller->ActiveGOAPPlan.Count)
41 {
42 Controller->bGOAPPlanActive = false;
43 return true;
44 }
45 Controller->ActiveGOAPPlan.Steps[Step]->Execute(Ctx);
46 Controller->GOAPPlanStep++;
47 if (Controller->GOAPPlanStep >= Controller->ActiveGOAPPlan.Count)
48 {
49 Controller->bGOAPPlanActive = false;
50 return true;
51 }
52 return false;
53 }
54 else // HTN
55 {
56 if (!Controller->ActiveHTNPlan.bValid) return true;
57 const uint32 Step = Controller->HTNPlanStep;
58 if (Step >= Controller->ActiveHTNPlan.Count)
59 {
60 Controller->bHTNPlanActive = false;
61 return true;
62 }
63 Controller->ActiveHTNPlan.Steps[Step]->Execute(Ctx);
64 Controller->HTNPlanStep++;
65 if (Controller->HTNPlanStep >= Controller->ActiveHTNPlan.Count)
66 {
67 Controller->bHTNPlanActive = false;
68 return true;
69 }
70 return false;
71 }
72}

74EBTNodeResult::Type UTactixBTTask_ExecutePlan::ExecuteTask(UBehaviorTreeComponent& OwnerComp,
75 uint8* NodeMemory)
76{
77 ATactixAIController* Controller = Cast<ATactixAIController>(OwnerComp.GetAIOwner());
78 if (!Controller) return EBTNodeResult::Failed;
79
80 const bool bGOAP = (PlanType == ETactixPlanType::GOAP);
81 if (bGOAP && (!Controller->ActiveGOAPPlan.bValid || Controller->ActiveGOAPPlan.Count == 0))
82 return EBTNodeResult::Failed;
83 if (!bGOAP && (!Controller->ActiveHTNPlan.bValid || Controller->ActiveHTNPlan.Count == 0))
84 return EBTNodeResult::Failed;
85
86 const bool bDone = StepPlan(OwnerComp);
87 return bDone ? EBTNodeResult::Succeeded : EBTNodeResult::InProgress;
88}
