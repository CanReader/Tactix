34void FTactixAgentProfiler::RecordSample(Tactix::FTactixHandle<Tactix::ITactixAgent> Agent,
35 FName Context, uint64 Start)
36{
37 const uint64 End = FPlatformTime::Cycles64();
38 const double ElapsedUs = static_cast<double>(End - Start)
39 * FPlatformTime::GetSecondsPerCycle64() * 1.0e6;
40
41 FTactixTimingSample Sample;
42 Sample.AgentHandle = Agent;
43 Sample.Context = Context;
44 Sample.ElapsedMicroseconds = ElapsedUs;
45 Sample.WallTimeSeconds = FPlatformTime::Seconds();
46
47 if (Ring.Num() < kRingSize)
48 {
49 Ring.Add(Sample);
50 }
51 else
52 {
53 Ring[Head] = Sample;
54 }
55 Head = (Head + 1) % kRingSize;
56 Count = FMath::Min(Count + 1, kRingSize);
57}

61double FTactixAgentProfiler::GetWorstCase(Tactix::FTactixHandle<Tactix::ITactixAgent> Agent,
62 FName Context) const
63{
64 double Worst = 0.0;
65 for (const FTactixTimingSample& S : Ring)
66 {
67 if (S.AgentHandle != Agent) continue;
68 if (!Context.IsNone() && S.Context != Context) continue;
69 Worst = FMath::Max(Worst, S.ElapsedMicroseconds);
70 }
71 return Worst;
72}

74double FTactixAgentProfiler::GetAverage(Tactix::FTactixHandle<Tactix::ITactixAgent> Agent,
75 FName Context) const
76{
77 double Sum = 0.0;
78 int32 N = 0;
79 for (const FTactixTimingSample& S : Ring)
80 {
81 if (S.AgentHandle != Agent) continue;
82 if (!Context.IsNone() && S.Context != Context) continue;
83 Sum += S.ElapsedMicroseconds;
84 ++N;
85 }
86 return (N > 0) ? Sum / static_cast<double>(N) : 0.0;
87}

89void FTactixAgentProfiler::Report(int32 TopN) const
90{
91 // Build a map of AgentHandle → worst-case for a ranked log dump.
92 TMap<uint32, double> WorstPerAgent;
93 for (const FTactixTimingSample& S : Ring)
94 {
95 const uint32 Key = S.AgentHandle.Packed;
96 double& W = WorstPerAgent.FindOrAdd(Key, 0.0);
97 W = FMath::Max(W, S.ElapsedMicroseconds);
98 }
99
100 // Sort descending.
101 WorstPerAgent.ValueSort([](double A, double B) { return A > B; });
102
103 UE_LOG(LogTactix, Log, TEXT("--- TactixAgentProfiler Report (top %d) ---"), TopN);
104 int32 Rank = 0;
105 for (const auto& Pair : WorstPerAgent)
106 {
107 UE_LOG(LogTactix, Log, TEXT(" [%d] Handle=0x%08X WorstCase=%.1f µs"),
108 ++Rank, Pair.Key, Pair.Value);
109 if (Rank >= TopN) break;
110 }
111}
