20void UTactixEQSTest_Influence::RunTest(FEnvQueryInstance& QueryInstance) const
21{
22 UObject* QueryOwner = QueryInstance.Owner.Get();
23 if (!QueryOwner) return;
24
25 UWorld* World = QueryOwner->GetWorld();
26 if (!World) return;
27
28 UTactixWorldSubsystem* TactixSub = World->GetSubsystem<UTactixWorldSubsystem>();
29 if (!TactixSub) return;
30
31 const Tactix::FTactixInfluenceMap<4>& Map = TactixSub->GetInfluenceMap();
32 const std::size_t Ch = static_cast<std::size_t>(FMath::Clamp(ChannelIndex, 0, 3));
33
34 for (FEnvQueryInstance::ItemIterator It(this, QueryInstance); It; ++It)
35 {
36 const FVector ItemLoc = QueryInstance.GetItemAsLocation(It.GetIndex());
37 const Tactix::FTactixVec2 XY{ (float)ItemLoc.X, (float)ItemLoc.Y };
38
39 float Score = bBilinear ? Map.SampleBilinear(Ch, XY) : Map.SampleNearest(Ch, XY);
40 if (bInvertScore) Score = 1.0f - FMath::Clamp(Score, 0.0f, 1.0f);
41
42 It.SetScore(TestPurpose, FilterType, Score, FloatValueMin.GetValue(), FloatValueMax.GetValue());
43 }
44}
