37 struct FOpenSet
38 {
39 int32_t* Heap;
40 uint32_t Size;
41 uint32_t Capacity;
42 const FNode* Nodes;
43
45 bool Push(int32_t Idx)
46 {
47 if (Size >= Capacity) return false;
48 uint32_t i = Size++;
49 Heap[i] = Idx;
50 while (i > 0)
51 {
52 const uint32_t p = (i - 1) >> 1;
53 if (Nodes[Heap[p]].F <= Nodes[Heap[i]].F) break;
54 const int32_t t = Heap[p]; Heap[p] = Heap[i]; Heap[i] = t;
55 i = p;
56 }
57 return true;
58 }
59
61 int32_t Pop()
62 {
63 if (Size == 0) return -1;
64 const int32_t Top = Heap[0];
65 Heap[0] = Heap[--Size];
66 uint32_t i = 0;
67 while (true)
68 {
69 const uint32_t l = 2u * i + 1u;
70 const uint32_t r = 2u * i + 2u;
71 uint32_t m = i;
72 if (l < Size && Nodes[Heap[l]].F < Nodes[Heap[m]].F) m = l;
73 if (r < Size && Nodes[Heap[r]].F < Nodes[Heap[m]].F) m = r;
74 if (m == i) break;
75 const int32_t t = Heap[i]; Heap[i] = Heap[m]; Heap[m] = t;
76 i = m;
77 }
78 return Top;
79 }
80 };

91 struct FClosedSet
92 {
93 int32_t* Slots;
94 uint32_t Capacity;
95 const FNode* Nodes;
96
98 int32_t Find(const FTactixWorldState& S) const
99 {
100 const uint64_t H = S.Hash();
101 const uint32_t Mask = Capacity - 1;
102 for (uint32_t p = 0; p < Capacity; ++p)
103 {
104 const uint32_t Slot = static_cast<uint32_t>((H + p) & Mask);
105 const int32_t Idx = Slots[Slot];
106 if (Idx < 0) return -1;
107 if (Nodes[Idx].State == S) return Idx;
108 }
109 return -1;
110 }
111
113 void InsertOrUpdate(int32_t Idx)
114 {
115 const FTactixWorldState& S = Nodes[Idx].State;
116 const uint64_t H = S.Hash();
117 const uint32_t Mask = Capacity - 1;
118 for (uint32_t p = 0; p < Capacity; ++p)
119 {
120 const uint32_t Slot = static_cast<uint32_t>((H + p) & Mask);
121 const int32_t Cur = Slots[Slot];
122 if (Cur < 0 || Nodes[Cur].State == S)
123 {
124 Slots[Slot] = Idx;
125 return;
126 }
127 }
128 }
129 };

132 constexpr uint32_t NextPow2(uint32_t V)
133 {
134 if (V <= 1) return 1;
135 V--; V |= V >> 1; V |= V >> 2; V |= V >> 4; V |= V >> 8; V |= V >> 16;
136 return V + 1;
137 }

140 FTactixGOAPPlan FTactixGOAPPlanner::Plan(
141 const FTactixWorldState& Start,
142 const FTactixWorldState& Goal,
143 ITactixGOAPAction* const* Actions,
144 uint32_t ActionCount,
145 const FTactixAgentContext& Ctx,
146 FTactixArena& Scratch,
147 FTactixGOAPConfig Config)
148 {
149 FTactixGOAPPlan Result{};
150
151 if (Config.MaxNodes == 0 || Config.MaxPlanLength == 0) return Result;
152
153 // Already satisfied -> empty plan, bValid=true, zero cost.
154 if (Start.Satisfies(Goal)) { Result.bValid = true; return Result; }
155
156 FNode* Nodes = Scratch.EmplaceArrayUninitialised<FNode> (Config.MaxNodes);
157 int32_t* Heap = Scratch.EmplaceArrayUninitialised<int32_t>(Config.MaxNodes);
158 if (Nodes == nullptr || Heap == nullptr) return Result;
159
160 const uint32_t ClosedCap = NextPow2(Config.MaxNodes * 2u);
161 int32_t* Closed = Scratch.EmplaceArrayUninitialised<int32_t>(ClosedCap);
162 if (Closed == nullptr) return Result;
163 for (uint32_t i = 0; i < ClosedCap; ++i) Closed[i] = -1;
164
165 FOpenSet Open { Heap, 0u, Config.MaxNodes, Nodes };
166 FClosedSet ClosedSet { Closed, ClosedCap, Nodes };
167
168 uint32_t NodeCount = 0;
169 Nodes[NodeCount] = { Start, -1, -1, 0.0f, static_cast<float>(Start.Distance(Goal)) };
170 ClosedSet.InsertOrUpdate(static_cast<int32_t>(NodeCount));
171 Open.Push(static_cast<int32_t>(NodeCount));
172 ++NodeCount;
173
174 int32_t GoalIdx = -1;
175
176 while (Open.Size > 0)
177 {
178 const int32_t CurIdx = Open.Pop();
179 const FNode Cur = Nodes[CurIdx];
180
181 if (Cur.State.Satisfies(Goal)) { GoalIdx = CurIdx; break; }
182
183 // Stale entry: a better node for this state was found after we pushed.
184 if (ClosedSet.Find(Cur.State) != CurIdx) continue;
185
186 for (uint32_t a = 0; a < ActionCount; ++a)
187 {
188 ITactixGOAPAction* A = Actions[a];
189 if (A == nullptr) continue;
190 if (!A->IsApplicable(Ctx)) continue;
191
192 if (!Cur.State.Satisfies(A->GetPreconditions(Ctx))) continue;
193
194 const float Cost = A->GetCost(Ctx);
195 if (Cost < 0.0f) continue;
196
197 FTactixWorldState Next = Cur.State;
198 Next.Apply(A->GetEffects(Ctx));
199
200 const float NewG = Cur.G + Cost;
201
202 const int32_t Existing = ClosedSet.Find(Next);
203 if (Existing >= 0 && Nodes[Existing].G <= NewG) continue;
204
205 if (NodeCount >= Config.MaxNodes) return Result;
206
207 const float NewF = NewG + static_cast<float>(Next.Distance(Goal));
208 Nodes[NodeCount] = { Next, CurIdx, static_cast<int32_t>(a), NewG, NewF };
209 ClosedSet.InsertOrUpdate(static_cast<int32_t>(NodeCount));
210 if (!Open.Push(static_cast<int32_t>(NodeCount))) return Result;
211 ++NodeCount;
212 }
213 }
214
215 if (GoalIdx < 0) return Result;
216
217 // Reconstruct by walking parent links from the goal back to the start,
218 // counting first to size the step array, then filling it back-to-front.
219 uint32_t Len = 0;
220 for (int32_t i = GoalIdx; i >= 0 && Nodes[i].ActionIdx >= 0; i = Nodes[i].Parent) ++Len;
221 if (Len == 0 || Len > Config.MaxPlanLength) return Result;
222
223 ITactixGOAPAction** Steps = Scratch.EmplaceArrayUninitialised<ITactixGOAPAction*>(Len);
224 if (Steps == nullptr) return Result;
225
226 uint32_t Write = Len;
227 for (int32_t i = GoalIdx; i >= 0 && Nodes[i].ActionIdx >= 0; i = Nodes[i].Parent)
228 {
229 Steps[--Write] = Actions[static_cast<uint32_t>(Nodes[i].ActionIdx)];
230 }
231
232 Result.Steps = Steps;
233 Result.Count = Len;
234 Result.Cost = Nodes[GoalIdx].G;
235 Result.bValid = true;
236 return Result;
237 }
