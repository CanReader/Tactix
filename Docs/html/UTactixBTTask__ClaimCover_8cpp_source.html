22EBTNodeResult::Type UTactixBTTask_ClaimCover::ExecuteTask(UBehaviorTreeComponent& OwnerComp,
23 uint8* NodeMemory)
24{
25 ATactixAIController* Controller = Cast<ATactixAIController>(OwnerComp.GetAIOwner());
26 if (!Controller) return EBTNodeResult::Failed;
27
28 UTactixAgentComponent* Agent = Controller->GetTactixAgent();
29 if (!Agent) return EBTNodeResult::Failed;
30
31 UWorld* World = OwnerComp.GetWorld();
32 if (!World) return EBTNodeResult::Failed;
33
34 UTactixWorldSubsystem* TactixSub = World->GetSubsystem<UTactixWorldSubsystem>();
35 if (!TactixSub) return EBTNodeResult::Failed;
36
37 const Tactix::FTactixAgentContext Ctx = Agent->BuildContext();
38
39 // Read threat location from Blackboard.
40 Tactix::FTactixVec3 ThreatPos{};
41 if (const UBlackboardComponent* BB = OwnerComp.GetBlackboardComponent())
42 {
43 const FVector BbThreat = BB->GetValueAsVector(ThreatLocationKey.SelectedKeyName);
44 ThreatPos = { (float)BbThreat.X, (float)BbThreat.Y, (float)BbThreat.Z };
45 }
46
47 Tactix::FTactixCoverSystem<512>::FQuery Q{};
48 Q.FromPosition = Ctx.Position;
49 Q.ThreatPosition = ThreatPos;
50 Q.MaxRange = MaxRange;
51 Q.MinBlockDot = MinBlockDot;
52 Q.bExcludeClaimed = bExcludeClaimed;
53 Q.Querier = Ctx.AgentHandle;
54
55 const auto CoverH = TactixSub->GetCoverSystem().QueryBest(Q);
56 if (!CoverH.IsValid()) return EBTNodeResult::Failed;
57
58 if (!TactixSub->GetCoverSystem().Claim(CoverH, Ctx.AgentHandle))
59 return EBTNodeResult::Failed;
60
61 // Mark the agent as in cover.
62 Agent->bInCover = true;
63
64 // Push the cover position to Blackboard if a key is configured.
65 if (UBlackboardComponent* BB = OwnerComp.GetBlackboardComponent())
66 {
67 if (const Tactix::FTactixCoverPoint* P = TactixSub->GetCoverSystem().Get(CoverH))
68 {
69 if (!CoverPositionKey.SelectedKeyName.IsNone())
70 {
71 BB->SetValueAsVector(CoverPositionKey.SelectedKeyName,
72 FVector(P->Position.X, P->Position.Y, P->Position.Z));
73 }
74 }
75 }
76
77 return EBTNodeResult::Succeeded;
78}
